<script>
    // --- DATA STRUCTURES ---
    const categories = ["Back", "Chest", "Biceps", "Shoulder", "Triceps", "Abs", "Legs"];
    
    let exercises = JSON.parse(localStorage.getItem('gymFlow_exercises')) || {
        "Back": ["Pull-up", "Lat Pulldown", "Barbell Row", "Deadlift", "Seated Cable Row"],
        "Chest": ["Bench Press", "Incline DB Press", "Chest Fly", "Push-up"],
        "Biceps": ["Barbell Curl", "Hammer Curl", "Preacher Curl"],
        "Shoulder": ["Overhead Press", "Lateral Raise", "Rear Delt Fly"],
        "Triceps": ["Tricep Pushdown", "Skull Crusher", "Dips"],
        "Abs": ["Plank", "Leg Raise", "Cable Crunch"],
        "Legs": ["Squat", "Leg Press", "Romanian Deadlift", "Calf Raise"]
    };

    let logs = JSON.parse(localStorage.getItem('gymFlow_logs')) || [];

    // --- INITIALIZATION ---
    function init() {
        renderCategories();
        updateStats();
        populateWeekFilter();
        renderHistory();
    }

    // --- NAVIGATION ---
    function showSection(id) {
        document.querySelectorAll('section').forEach(s => s.classList.add('hidden'));
        document.getElementById(id).classList.remove('hidden');
        document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
        event.target.classList.add('active');
    }

    // --- RENDERING ---
    function renderCategories() {
        const grid = document.getElementById('categoryGrid');
        grid.innerHTML = categories.map(cat => `
            <div class="cat-card" onclick="openCategory('${cat}')">
                <i>${getEmoji(cat)}</i>
                <strong>${cat}</strong>
            </div>
        `).join('');
    }

    function openCategory(cat) {
        showSection('workout');
        document.getElementById('currentCatTitle').innerText = `${cat} Exercises`;
        const list = document.getElementById('exerciseList');
        list.innerHTML = exercises[cat].map(ex => `
            <div class="cat-card" style="padding: 15px; border-style: dashed;">
                <p style="font-weight: 600; margin-bottom: 10px;">${ex}</p>
                <div style="display: flex; gap: 5px;">
                    <button onclick="openLogModal('${cat}', '${ex}')" class="tab-btn" style="background: var(--accent); color:white; flex:1; padding: 5px;">Log</button>
                    <button onclick="openHelp('${ex}', '${cat}')" class="tab-btn" style="background: var(--border); flex:0.4; padding: 5px;">?</button>
                </div>
            </div>
        `).join('');
    }

    // --- LOGGING LOGIC ---
    function openLogModal(cat, ex) {
        const modal = document.getElementById('logModal');
        document.getElementById('modalExName').innerText = ex;
        document.getElementById('logExName').value = ex;
        document.getElementById('logCat').value = cat;
        
        // Find last performance
        const lastLog = logs.findLast(l => l.exercise === ex);
        document.getElementById('prevPerformance').innerText = lastLog ? 
            `Last time: ${lastLog.weight}kg for ${lastLog.sets}x${lastLog.reps}` : "No previous data";

        modal.style.display = 'flex';
    }

    document.getElementById('logForm').onsubmit = (e) => {
        e.preventDefault();
        const newLog = {
            id: Date.now(),
            date: new Date().toLocaleDateString(),
            week: getWeekNumber(new Date()),
            category: document.getElementById('logCat').value,
            exercise: document.getElementById('logExName').value,
            sets: document.getElementById('logSets').value,
            reps: document.getElementById('logReps').value,
            weight: document.getElementById('logWeight').value,
            notes: document.getElementById('logNotes').value
        };
        logs.unshift(newLog);
        saveData();
        closeModal('logModal');
        updateStats();
        renderHistory();
        alert('Workout Logged!');
    };

    // --- HISTORY & STATS ---
    function renderHistory() {
        const query = document.getElementById('searchLog').value.toLowerCase();
        const weekFilter = document.getElementById('filterWeek').value;
        const body = document.getElementById('historyBody');
        
        const filtered = logs.filter(l => {
            const matchesSearch = l.exercise.toLowerCase().includes(query);
            const matchesWeek = weekFilter === "" || l.week == weekFilter;
            return matchesSearch && matchesWeek;
        });

        body.innerHTML = filtered.map(l => `
            <tr>
                <td><small>${l.date}</small></td>
                <td><strong>${l.exercise}</strong><br><small>${l.category}</small></td>
                <td>${l.sets} x ${l.reps}</td>
                <td>${l.weight}kg</td>
                <td><button onclick="deleteLog(${l.id})" style="color:var(--danger); border:none; background:none; cursor:pointer;">Delete</button></td>
            </tr>
        `).join('');
    }

    function deleteLog(id) {
        if(confirm('Delete this entry?')) {
            logs = logs.filter(l => l.id !== id);
            saveData();
            renderHistory();
            updateStats();
        }
    }

    function updateStats() {
        const currentWeek = getWeekNumber(new Date());
        const weekLogs = logs.filter(l => l.week === currentWeek);
        
        // Count unique dates this week
        const uniqueDays = new Set(weekLogs.map(l => l.date)).size;
        document.getElementById('statWorkouts').innerText = uniqueDays;

        // Calculate Volume
        const volume = weekLogs.reduce((acc, curr) => acc + (curr.sets * curr.reps * curr.weight), 0);
        document.getElementById('statVolume').innerText = volume.toLocaleString();
    }

    // --- HELPERS ---
    function getEmoji(cat) {
        const emojis = { "Back": "🛶", "Chest": "🦍", "Biceps": "💪", "Shoulder": "🛡️", "Triceps": "🏹", "Abs": "🧱", "Legs": "🦵" };
        return emojis[cat] || "🏋️";
    }

    function getWeekNumber(d) {
        d = new Date(Date.UTC(d.getFullYear(), d.getMonth(), d.getDate()));
        d.setUTCDate(d.getUTCDate() + 4 - (d.getUTCDay() || 7));
        var yearStart = new Date(Date.UTC(d.getUTCFullYear(), 0, 1));
        return Math.ceil((((d - yearStart) / 86400000) + 1) / 7);
    }

    function populateWeekFilter() {
        const weeks = [...new Set(logs.map(l => l.week))].sort((a,b) => b-a);
        const select = document.getElementById('filterWeek');
        select.innerHTML = '<option value="">All Weeks</option>' + 
            weeks.map(w => `<option value="${w}">Week ${w}</option>`).join('');
    }

    function openHelp(ex, cat) {
        const modal = document.getElementById('helpModal');
        const content = document.getElementById('helpContent');
        content.innerHTML = `
            <span class="tag">${cat}</span>
            <h2>${ex}</h2>
            <div style="background:#eee; height:200px; display:flex; align-items:center; justify-content:center; border-radius:12px; margin:15px 0; color:#666;">
                
            </div>
            <h3>Instructions</h3>
            <p>1. Maintain a neutral spine and engage your core.<br>
               2. Perform the movement through a full range of motion.<br>
               3. Control the weight on the eccentric (downward) phase.</p>
            <br>
            <h3>Target Muscle</h3>
            <p>Primary: ${cat}</p>
        `;
        modal.style.display = 'flex';
    }

    function openCustomModal() { document.getElementById('customModal').style.display = 'flex'; }

    function saveCustomExercise() {
        const cat = document.getElementById('customCat').value;
        const name = document.getElementById('customName').value;
        if(name) {
            exercises[cat].push(name);
            localStorage.setItem('gymFlow_exercises', JSON.stringify(exercises));
            openCategory(cat);
            closeModal('customModal');
            document.getElementById('customName').value = "";
        }
    }

    function closeModal(id) { document.getElementById(id).style.display = 'none'; }

    function saveData() { localStorage.setItem('gymFlow_logs', JSON.stringify(logs)); }

    // Theme Toggle
    document.getElementById('themeToggle').onclick = () => {
        const current = document.documentElement.getAttribute('data-theme');
        document.documentElement.setAttribute('data-theme', current === 'dark' ? 'light' : 'dark');
    };

    window.onclick = (event) => {
        if (event.target.className === 'modal') closeModal(event.target.id);
    };

    init();
</script>
